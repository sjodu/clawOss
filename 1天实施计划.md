## 阶段 2: 预算控制（2 小时）

### 2.1 预算监控脚本（30 分钟）

创建 `scripts/budget-status.sh`:

```bash
#!/bin/bash

BUDGET_MAX=$(cat ~/.openclaw/budget-max.txt 2>/dev/null || echo "100.0")
BUDGET_SPENT=$(cat ~/.openclaw/budget-spent.txt 2>/dev/null || echo "0.0")
BUDGET_REMAINING=$(echo "$BUDGET_MAX - $BUDGET_SPENT" | bc)
BUDGET_PERCENT=$(echo "scale=2; $BUDGET_SPENT / $BUDGET_MAX * 100" | bc)

echo "╔════════════════════════════════════════╗"
echo "║        BUDGET STATUS                   ║"
echo "╠════════════════════════════════════════╣"
printf "║ Max Budget:      \$%-18s ║\n" "$BUDGET_MAX"
printf "║ Spent:           \$%-18s ║\n" "$BUDGET_SPENT"
printf "║ Remaining:       \$%-18s ║\n" "$BUDGET_REMAINING"
printf "║ Usage:           %-18s%% ║\n" "$BUDGET_PERCENT"
echo "╚════════════════════════════════════════╝"

if (( $(echo "$BUDGET_SPENT >= $BUDGET_MAX" | bc -l) )); then
  echo ""
  echo "⚠️  BUDGET EXCEEDED - Gateway stopped"
  echo "To reset: echo '0.0' > ~/.openclaw/budget-spent.txt"
elif (( $(echo "$BUDGET_SPENT >= $BUDGET_MAX * 0.8" | bc -l) )); then
  echo ""
  echo "⚠️  WARNING: 80% budget used"
fi
```

### 2.3 集成到 Dashboard（1 小时 15 分钟）

#### 修改 `dashboard/app/api/ingest/metrics/route.ts`:

```typescript
// 添加预算信息到响应
export async function POST(request: Request) {
  const body = await request.json();
  
  // 保存 metrics（原有逻辑）
  await db.insert(metrics).values({
    inputTokens: body.inputTokens,
    outputTokens: body.outputTokens,
    costUsd: body.costUsd,
    timestamp: new Date()
  });
  
  // 计算预算状态
  const totalSpent = await db
    .select({ sum: sql`SUM(cost_usd)` })
    .from(metrics);
  
  const budgetMax = parseFloat(process.env.BUDGET_MAX_USD || '100');
  const budgetSpent = totalSpent[0]?.sum || 0;
  const budgetRemaining = budgetMax - budgetSpent;
  
  return Response.json({
    success: true,
    budget: {
      max: budgetMax,
      spent: budgetSpent,
      remaining: budgetRemaining,
      percent: (budgetSpent / budgetMax) * 100
    }
  });
}
```
